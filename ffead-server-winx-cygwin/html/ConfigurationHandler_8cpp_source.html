00035 void ConfigurationHandler::listi(string cwd,string type,bool apDir,strVec &folders)
00036 {
00037         Logger logger = Logger::getLogger("ConfigurationHandler");
00038         FILE *pipe_fp;
00039         string command;
00040         command = ("ls -F1 "+cwd+"|grep '"+type+"'");
00041         logger << "\nCommand:" << command << flush;
00042         if ((pipe_fp = popen(command.c_str(), "r")) == NULL)
00043         {
00044                 printf("pipe open error in cmd_list\n");
00045         }
00046         int t_char;
00047         string folderName;
00048         while ((t_char = fgetc(pipe_fp)) != EOF)
00049         {
00050                 if(t_char!='\n')
00051                 {
00052                         stringstream ss;
00053                         ss << (char)t_char;
00054                         string temp;
00055                         ss >> temp;
00056                         folderName.append(temp);
00057                 }
00058                 else if(folderName!="")
00059                 {
00060                         StringUtil::replaceFirst(folderName,"*","");
00061                         if(folderName.find("~")==string::npos)
00062                         {
00063                                 logger << "\nlist for file" << (cwd+"/"+folderName) << "\n" << flush;
00064                                 if(apDir)
00065                                         folders.push_back(cwd+folderName);
00066                                 else
00067                                         folders.push_back(folderName);
00068                         }
00069                         folderName = "";
00070                 }
00071         }
00072         pclose(pipe_fp);
00073 }
