00046 void ComponentHandler::service(int fd)
00047 {
00048         init();
00049         string methInfo,retValue;
00050         _cmp_instance->getServer()->Receive(fd,methInfo,1024);
00051         methInfo =methInfo.substr(0,methInfo.find_last_of(">")+1);
00052         _cmp_instance->logger << methInfo << flush;
00053         try
00054         {
00055                 XmlParser parser("Parser");
00056                 _cmp_instance->logger << "\nBean call parsed successfully\n" << flush;
00057                 if(methInfo.find("lang=\"c++\"")!=string::npos || methInfo.find("lang='c++'")!=string::npos)
00058                 {
00059                         Document doc = parser.getDocument(methInfo);
00060                         Element message = doc.getRootElement();
00061                         if(message.getTagName()!="service")
00062                         {
00063                                 throw new ComponentHandlerException("No service Tag\n",retValue);
00064                         }
00065                         if(message.getAttributes().size()<4)
00066                         {
00067                                 throw new ComponentHandlerException("name,beanName,returnType and lang are mandatory attributes\n",retValue);
00068                         }
00069                         else if(message.getAttribute("name")=="")
00070                         {
00071                                 throw new ComponentHandlerException("name attribute missing\n",retValue);
00072                         }
00073                         else if(message.getAttribute("returnType")=="")
00074                         {
00075                                 throw new ComponentHandlerException("returnType attribute missing\n",retValue);
00076                         }
00077                         else if(message.getAttribute("beanName")=="")
00078                         {
00079                                 throw new ComponentHandlerException("beanName attribute missing\n",retValue);
00080                         }
00081                         else if(message.getAttribute("lang")=="")
00082                         {
00083                                 throw new ComponentHandlerException("lang attribute missing\n",retValue);
00084                         }
00085                         if(message.getChildElements().size()!=1)
00086                         {
00087                                 throw new ComponentHandlerException("message tag should have only one child tag\n",retValue);
00088                         }
00089                         else if(message.getChildElements().at(0).getTagName()!="args")
00090                         {
00091                                 throw new ComponentHandlerException("message tag should have an args child tag\n",retValue);
00092                         }
00093                         Serialize ser;
00094                         Reflector reflector;
00095                         args argus;
00096                         vals valus;
00097                         ElementList argts = message.getChildElements().at(0).getChildElements();
00098                         for (unsigned var = 0; var < argts.size();  var++)
00099                         {
00100                                 void *value = NULL;
00101                                 Element arg = argts.at(var);
00102                                 if(arg.getTagName()!="argument")
00103                                         throw new ComponentHandlerException("Invalid Tag, only argument tag allowed\n",retValue);
00104                                 else if(arg.getAttribute("type")=="")
00105                                         throw new ComponentHandlerException("every argument tag should have a type attribute\n",retValue);
00106                                 if(arg.getText()=="" && arg.getChildElements().size()==0)
00107                                         throw new ComponentHandlerException("argument value missing\n",retValue);
00108                                 if(arg.getAttribute("type")!="")
00109                                 {
00110                                         Element obj = arg.getChildElements().at(0);
00111                                         string objxml = obj.render();
00112                                         string objClassName = obj.getTagName();
00113                                         value = ser.unSerializeUnknown(objxml,arg.getAttribute("type"));
00114                                 }
00115                                 argus.push_back(arg.getAttribute("type"));
00116                                 valus.push_back(value);
00117                         }
00118                         string className = "Component_"+message.getAttribute("beanName");
00119                         _cmp_instance->logger << "\nBean class = " << className << "\n" << flush;
00120                         string returnType = message.getAttribute("returnType");
00121                         string lang = message.getAttribute("lang");
00122                         ClassInfo clas = reflector.getClassInfo(className);
00123                         string methodName = message.getAttribute("name");
00124                         _cmp_instance->logger << "\nBean service = " << methodName << "\n" << flush;
00125                         if(clas.getClassName()=="")
00126                         {
00127                                 throw new ComponentHandlerException("bean does not exist or is not regsitered\n",retValue);
00128                         }
00129                         Method meth = clas.getMethod(methodName,argus);
00130                         if(meth.getMethodName()=="")
00131                         {
00132                                 throw new ComponentHandlerException("service does not exist for the bean or the bean does not exist or is not regsitered\n\n",retValue);
00133                         }
00134                         else
00135                         {
00136                                 _cmp_instance->logger << "\nGot Bean service " << methodName << "\n" << flush;
00137                                 args argus;
00138                                 Constructor ctor = clas.getConstructor(argus);
00139                                 void *_temp = reflector.newInstanceGVP(ctor);
00140                                 _cmp_instance->logger << "\nGot Bean " << _temp << "\n" << flush;
00141                                 if(returnType=="void" || returnType=="")
00142                                 {
00143                                         _cmp_instance->logger << "\nVoid return " << "\n" << flush;
00144                                         reflector.invokeMethod<void*>(_temp,meth,valus);
00145                                         retValue = ("<return:void></return:void>");
00146                                 }
00147                                 else
00148                                 {
00149                                         _cmp_instance->logger << "\nReturn type = "<< returnType << "\n" << flush;
00150                                         if(returnType=="int")
00151                                         {
00152                                                 int retv = reflector.invokeMethod<int>(_temp,meth,valus);
00153                                                 retValue = ("<return:int>"+boost::lexical_cast<string>(retv)+"</return:int>");
00154                                         }
00155                                         else if(returnType=="float")
00156                                         {
00157                                                 float retv = reflector.invokeMethod<float>(_temp,meth,valus);
00158                                                 retValue = ("<return:float>"+boost::lexical_cast<string>(retv)+"</return:float>");
00159                                         }
00160                                         else if(returnType=="double")
00161                                         {
00162                                                 double retv = reflector.invokeMethod<double>(_temp,meth,valus);
00163                                                 retValue = ("<return:double>"+boost::lexical_cast<string>(retv)+"</return:double>");
00164                                         }
00165                                         else if(returnType=="string")
00166                                         {
00167                                                 string retv = reflector.invokeMethod<string>(_temp,meth,valus);
00168                                                 retValue = ("<return:string>"+retv+"</return:string>");
00169                                         }
00170                                         else if(returnType!="")
00171                                         {
00172                                                 void* retobj = reflector.invokeMethodUnknownReturn(_temp,meth,valus);
00173                                                 string oxml = ser.serializeUnknown(retobj,returnType);
00174                                                 retValue = ("<return:"+returnType+">"+oxml+"</return:"+returnType+">");
00175                                         }
00176                                 }
00177 
00178                         }
00179                 }
00180                 else
00181                 {
00182                         retValue = "<return:exception>This is a C++ daemon</return:exception>";
00183                 }
00184                 _cmp_instance->logger << "\nSending data = "<< retValue << "\n" << flush;
00185                 if(retValue!="")
00186                         _cmp_instance->getServer()->Send(fd,retValue);
00187                 //close(fd);
00188         }
00189         catch(ComponentHandlerException *e)
00190         {
00191                 _cmp_instance->logger << e->getMessage() << flush;
00192                 _cmp_instance->getServer()->Send(fd,retValue);
00193                 close(fd);
00194         }
00195         catch(...)
00196         {
00197                 _cmp_instance->logger << "exception occurred" << flush;
00198                 retValue = ("<return:exception>XmlParseException occurred</return:exception>");
00199                 _cmp_instance->getServer()->Send(fd,retValue);
00200                 close(fd);
00201         }
00202 }
