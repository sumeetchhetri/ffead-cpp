00185     string generateResponse()
00186     {
00187         string resp;
00188         resp = (httpVersion + " " + statusCode + " " + statusMsg + "\r\n");
00189         if(this->server!="")resp += "Server: " + this->server + "\r\n";
00190         if(this->date!="")resp += "Date: " + this->date + "\r\n";
00191         if(this->connection!="")resp += "Connection: " + this->connection + "\r\n";
00192         if(this->location!="")resp += "Location: " + this->location + "\r\n";
00193         if(this->accept_ranges!="")resp += "Accept-Ranges: " + this->accept_ranges + "\r\n";
00194         if(this->content_type!="")resp += "Content-Type: " + this->content_type + "\r\n";
00195         if(this->content_str!="")resp += "Content-Length: " + boost::lexical_cast<string>(content_str.length()) + "\r\n";
00196         if(this->last_modified!="")resp += "Last-Modified: " + this->last_modified + "\r\n";
00197         for (int var = 0; var < (int)this->cookies.size(); var++)
00198         {
00199                 resp += "Set-Cookie: " + this->cookies.at(var) + "\r\n";
00200                 }
00201         resp += "\r\n";
00202         resp += this->content_str;
00203         return resp;
00204     }
