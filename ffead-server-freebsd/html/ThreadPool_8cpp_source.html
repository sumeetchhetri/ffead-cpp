00123 void ThreadPool::poll() {
00124         while (true) {
00125                 if (!prioritypooling) {
00126                         if (wpool->tasksPending()) {
00127                                 Task *task = wpool->getTask();
00128                                 Thread *idleThread = getIdleThread();
00129                                 idleThread->task = task;
00130                                 idleThread->idle = false;
00131                                 idleThread->mthread->interrupt();
00132                         } else {
00133                                 boost::this_thread::sleep(boost::posix_time::milliseconds(1));
00134                         }
00135                 } else {
00136                         if (wpool->tasksPPending()) {
00137                                 Task *task = wpool->getPTask();
00138                                 Thread *idleThread = getIdleThread();
00139                                 idleThread->task = task;
00140                                 idleThread->idle = false;
00141                                 idleThread->mthread->interrupt();
00142                         } else {
00143                                 boost::this_thread::sleep(boost::posix_time::milliseconds(1));
00144                         }
00145                 }
00146         }

00148 Thread* ThreadPool::getIdleThread() {
00149         while (true) {

00151                         if (tpool->at(var)->idle) {

00154                 }
00155                 boost::this_thread::sleep(boost::posix_time::milliseconds(1));
00156         }

00158 void ThreadPool::joinAll() {
00159         while (true) {
00160                 if (!prioritypooling) {
00161                         while (wpool->tasksPending()) {
00162                                 boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
00163                         }
00164                 } else {
00165                         while (wpool->tasksPPending()) {
00166                                 boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
00167                         }
00168                 }
00169                 int i = 0;
00170                 for (unsigned int var = 0; var < tpool->size(); var++) {
00171                         if (tpool->at(var)->idle) {
00172                                 i++;
00173                         }
00174                 }
00175                 if (i == initThreads) {
00176                         break;
00177                 } else {
00178                         boost::this_thread::sleep(boost::posix_time::milliseconds(1000));
00179                 }
00180         }
