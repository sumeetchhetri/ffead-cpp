00034 char* CryptoHandler::base64decode(unsigned char *input, int length)
00035 {
00036         BIO *b64, *bmem;
00037 
00038         char *buffer = (char *)malloc(length);
00039         memset(buffer, 0, length);
00040 
00041         b64 = BIO_new(BIO_f_base64());
00042         BIO_set_flags(b64, BIO_FLAGS_BASE64_NO_NL);
00043         bmem = BIO_new_mem_buf(input, length);
00044         bmem = BIO_push(b64, bmem);
00045 
00046         BIO_read(bmem, buffer, length);
00047 
00048         BIO_free_all(bmem);
00049 
00050         return buffer;

00053 char* CryptoHandler::base64encode(const unsigned char *input, int length)
00054 {
00055         BIO *bmem, *b64;
00056         BUF_MEM *bptr;
00057 
00058         b64 = BIO_new(BIO_f_base64());
00059         bmem = BIO_new(BIO_s_mem());
00060         b64 = BIO_push(b64, bmem);
00061         BIO_write(b64, input, length);
00062         BIO_flush(b64);
00063         BIO_get_mem_ptr(b64, &bptr);
00064 
00065         char *buff = (char *)malloc(bptr->length);
00066         memcpy(buff, bptr->data, bptr->length-1);
00067         buff[bptr->length-1] = 0;
00068 
00069         BIO_free_all(b64);
00070 
00071         return buff;
00072 }
