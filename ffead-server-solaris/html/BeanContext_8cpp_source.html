00114                 {
00115                 }

00121                 {
00122                         string tag = message.getTagName();
00123                         StringUtil::replaceFirst(tag,"return:","");
00124                         message.setTagName(tag);
00125                         call = message.render();
00126                         logger << call << flush;
00127                         retval = ser.unSerializeUnknown(call,tag);
00128                 }
00129                 else
00130                 {
00131                         throw "Exception occurred";
